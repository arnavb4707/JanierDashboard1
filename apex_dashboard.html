Google Sheets Sync

Enter your Apps Script Web App URL to sync live data from your Google Sheet. Data is cached locally so the dashboard works offline.

Setup Guide

1
Open your Google Sheet → Extensions → Apps Script. Paste the code below and save:
function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const result = { john: [], prabh: [], calls: [] };

  const s1 = ss.getSheets()[0];
  const d1 = s1.getDataRange().getValues();
  for (let i = 1; i < d1.length; i++) {
    const r = d1[i];
    if (!r[0]) continue;
    result.john.push({
      name:r[0],date:r[1],leads:r[2],convos:r[3],sets:r[4],
      cal:r[5],live:r[6],closes:r[7],downsells:r[8],
      cash:r[9],rev:r[10],rating:r[11],objections:r[12],notes:r[13]
    });
  }

  const s2 = ss.getSheets()[1];
  const d2 = s2.getDataRange().getValues();
  for (let i = 1; i < d2.length; i++) {
    const r = d2[i];
    if (!r[0]) continue;
    result.prabh.push({
      name:r[0],date:r[1],dials:r[2],connections:r[3],convos:r[4],
      sets:r[5],cal:r[6],live:r[7],closes:r[8],downsells:r[9],
      cash:r[10],rev:r[11],rating:r[12],objections:r[13],notes:r[14]
    });
  }

  const s3 = ss.getSheets()[2];
  const d3 = s3.getDataRange().getValues();
  for (let i = 1; i < d3.length; i++) {
    const r = d3[i];
    if (!r[0]) continue;
    result.calls.push({
      closer:r[0],date:r[1],email:r[2],summary:r[3],
      offer:r[4],status:r[5],cash:r[6],rev:r[7],recording:r[8]
    });
  }

  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}
2
Click Deploy → New Deployment. Set type to Web App. Execute as Me. Who has access: Anyone. Click Deploy.
3
Copy the Web App URL and paste it into the field above. Click Sync Sheet to pull live data.
